CREATE TABLE questions(id SERIAL PRIMARY KEY, 
					   text VARCHAR(200) NOT NULL UNIQUE,
					   right_answer INT NOT NULL,  
					   FOREIGN KEY(right_answer) REFERENCES answers(id)
);

CREATE TABLE question_to_answer(question_to_answer_id SERIAL PRIMARY KEY, 
								question_id INT, 
								answer_id INT, 
							   FOREIGN KEY (question_id) REFERENCES questions(id), 
							   FOREIGN KEY (answer_id) REFERENCES answers(id));
